## Problem 1

``` r
set.seed(1)


# Function to simulate birthdays for n people

simulateBirthday = function(n) {
  birthdays = sample(1:365, n, replace = TRUE)
  hasDuplicate = any(duplicated(birthdays))
  tibble(hasDuplicate = hasDuplicate)
}


# Group sizes to consider
groupSizes = 2:50


# Number of simulations per group size
nSim = 10000


# Create a tibble with all iterations for all group sizes

simDf = expand_grid(
  groupSize = groupSizes,
  iter = 1:nSim
) |>
  mutate(simResult = map(groupSize, simulateBirthday)) |>
  unnest(simResult)



# Compute probability for each group size

birthdayProb = simDf |>
  group_by(groupSize) |>
  summarize(prob = mean(hasDuplicate))
```
